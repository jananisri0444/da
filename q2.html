import pandas as pd
import numpy as np

np.random.seed(42)

n = 1000
regions = ["North", "South", "East", "West"]
products = ["Electronics", "Clothing", "Furniture", "Groceries"]

df = pd.DataFrame({
    "Order_ID": np.arange(1, n+1),
    "Region": np.random.choice(regions, n),
    "Product_Category": np.random.choice(products, n),
    "Units_Sold": np.random.randint(1, 20, n),
    "Unit_Price": np.random.randint(100, 2000, n),
    "Discount_%": np.random.randint(0, 30, n)
})
df["Revenue"] = df["Units_Sold"] * df["Unit_Price"]
df["Discount_Amount"] = df["Revenue"] * (df["Discount_%"] / 100)
df["Net_Sales"] = df["Revenue"] - df["Discount_Amount"]
